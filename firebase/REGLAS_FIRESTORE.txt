rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAuth() {
      return request.auth != null;
    }

    function userDoc() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }

    function userExists() {
      return isAuth() && exists(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function isActive() {
      return userExists() && userDoc().activo == true;
    }

    function roleIs(roles) {
      return isActive() && userDoc().role in roles;
    }

    function isClient() {
      return isActive() && userDoc().role == "client";
    }

    function myClientId() {
      return userDoc().clienteId;
    }

    match /users/{uid} {
      allow read: if isAuth() && (request.auth.uid == uid || roleIs(["admin"]));
      allow create: if roleIs(["admin"]);
      allow update: if roleIs(["admin"]) || (isAuth() && request.auth.uid == uid);
      allow delete: if roleIs(["admin"]);
    }

    match /clientes/{id} {
      allow read: if isActive() || (isClient() && myClientId() == id);
      allow create, update: if roleIs(["admin", "comercial"]);
      allow delete: if roleIs(["admin"]);
    }

    match /clientInvites/{code} {
      allow read: if true;
      allow create: if roleIs(["admin", "comercial"]);
      allow update: if roleIs(["admin", "comercial"])
        || (isAuth()
            && resource.data.enabled == true
            && resource.data.claimedByUid == null
            && request.resource.data.claimedByUid == request.auth.uid);
      allow delete: if roleIs(["admin"]);
    }

    match /cotizaciones/{id} {
      allow read: if isActive()
        || (isClient() && resource.data.clienteId == myClientId());
      allow create, update: if roleIs(["admin", "comercial"]);
      allow delete: if roleIs(["admin"]);
    }

    match /publicQuotes/{id} {
      allow read: if resource.data.enabled == true;
      allow create, update: if roleIs(["admin", "comercial"]);
      allow delete: if roleIs(["admin"]);
    }

    match /proyectos/{id} {
      allow read: if isActive()
        || (isClient() && resource.data.clienteId == myClientId());
      allow create, update: if roleIs(["admin", "comercial", "tecnico"]);
      allow delete: if roleIs(["admin"]);
    }

    match /documentos/{id} {
      allow read: if isActive()
        || (isClient() && resource.data.clienteId == myClientId());
      allow create, update: if roleIs(["admin", "comercial", "tecnico"]);
      allow delete: if roleIs(["admin", "comercial"]);
    }

    match /contabilidad/{id} {
      allow read, write: if roleIs(["admin", "contador", "finanzas"]);
    }

    match /financiero/{id} {
      allow read, write: if roleIs(["admin", "contador", "finanzas"]);
    }

    match /empleados/{id} {
      allow read, write: if roleIs(["admin", "rrhh"]);
    }

    match /nomina/{id} {
      allow read, write: if roleIs(["admin", "rrhh"]);
    }

    match /cuentas/{id} {
      allow read: if isActive() || (isClient() && resource.data.clienteId == myClientId());
      allow write: if roleIs(["admin", "comercial", "contador", "finanzas"]);
    }
  }
}
